public static boolean isPrime(int n) {
	if (n <= 1) return false;       // 0, 1, 음수는 소수 아님
	if (n == 2) return true;        // 2는 소수
	if (n % 2 == 0) return false;   // 짝수는 소수 아님

	for (int i = 3; i <= Math.sqrt(n); i += 2) {
		if (n % i == 0) return false;
	}
	return true;
}
